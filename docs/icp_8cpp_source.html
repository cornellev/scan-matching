15 void ICP::begin(const std::vector<Vector>& a, const std::vector<Vector>& b,
16 RBTransform t) {
17 // Initial transform guess
18 this->transform = t;
19
20 // Copy in point clouds
21 this->a = a;
22 this->b = b;
23
24 // Set relative to centroid
25 a_cm = get_centroid(a);
26 b_cm = get_centroid(b);
27 for (Vector& point: this->a) {
28 point -= a_cm;
29 }
30 for (Vector& point: this->b) {
31 point -= b_cm;
32 }
33
34 // Cost is infinite
35 previous_cost = std::numeric_limits<double>::infinity();
36 current_cost = std::numeric_limits<double>::infinity();
37
38 // Ensure arrays are the right size
39 const size_t n = a.size();
40 pair.reserve(n);
41 dist.reserve(n);
42
43 // Per-instance customization routine
44 setup();
45 }

47 void ICP::converge(double convergence_threshold) {
48 while (current_cost > convergence_threshold
49 && (current_cost < previous_cost || current_cost == INFINITY)) {
50 previous_cost = current_cost;
51 iterate();
52 }
