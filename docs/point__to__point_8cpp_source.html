20 void iterate(const std::vector<icp::Vector>& a,
21 const std::vector<icp::Vector>& b) override {
22 size_t n = a.size();
23 size_t m = b.size();
24 pair.resize(n);
25 dist.resize(n);
26
27 // compute center of mass of a
28 double a_cm_x = 0;
29 double a_cm_y = 0;
30 for (size_t i = 0; i < n; i++) {
31 a_cm_x += a[i][0];
32 a_cm_y += a[i][1];
33 }
34 a_cm_x /= n;
35 a_cm_y /= n;
36
37 transform.cm[0] = a_cm_x;
38 transform.cm[1] = a_cm_y;
39
40 // point-to-point matching O(n^2)
41 for (size_t i = 0; i < n; i++) {
42 dist[i] = std::numeric_limits<double>::infinity();
43 for (size_t j = 0; j < m; j++) {
44 double alt = std::hypot(a[i][0] - b[j][0],
45 a[i][1] - b[j][1]);
46 if (alt < dist[i]) {
47 pair[i] = j;
48 dist[i] = alt;
49 }
50 }
51 }
52
53 // modified gradient descent
54 double dx_effect = 0;
55 double dy_effect = 0;
56 double theta_effect = 0;
57 for (size_t i = 0; i < n; i++) {
58 double index_dist = ((double)i - pair[i]) / 200;
59 double weight = 1 + 20 * std::exp(-index_dist + index_dist);
60 double o_x = (a[i][0] - a_cm_x);
61 double o_y = (a[i][1] - a_cm_y);
62 double a_rot_x = std::cos(transform.theta) * o_x
63 - std::sin(transform.theta) * o_y + a_cm_x;
64 double a_rot_y = std::sin(transform.theta) * o_x
65 + std::cos(transform.theta) * o_y + a_cm_y;
66 double b_x = b[pair[i]][0];
67 double b_y = b[pair[i]][1];
68 dx_effect += (a_rot_x + transform.dx - b_x) * 2 / n * weight;
69 dy_effect += (a_rot_y + transform.dy - b_y) * 2 / n * weight;
70 // if (dx_effect > 10000) {
71 // Log << a_rot_x << ' ' << t.dx << ' ' << b_x << '\n';
72 // std::exit(1);
73 // }
74 theta_effect +=
75 (b_x * o_x * std::sin(transform.theta)
76 + b_x * o_y * std::cos(transform.theta)
77 - b_y * o_x * std::cos(transform.theta)
78 + b_y * o_y * std::sin(transform.theta)
79 - a_cm_x * o_x * std::sin(transform.theta)
80 - a_cm_x * o_y * std::cos(transform.theta)
81 + a_cm_y * o_x * std::cos(transform.theta)
82 - a_cm_y * o_y * std::sin(transform.theta)
83 - o_x * transform.dx * std::sin(transform.theta)
84 + o_x * transform.dy * std::cos(transform.theta)
85 - o_y * transform.dx * std::cos(transform.theta)
86 - o_y * transform.dy * std::sin(transform.theta))
87 * 2 / n * weight;
88 }
89 // Log << "dx_effect * rate = " << (dx_effect * rate) << '\n';
90 // Log << "dy_effect * rate = " << (dy_effect * rate) << '\n';
91 // Log << "dt_effect * rate = " << (theta_effect * rate) << '\n';
92 // std::exit(1);
93 transform.dx -= dx_effect * rate + momentum_alpha * last_dx_effect;
94 transform.dy -= dy_effect * rate + momentum_alpha * last_dy_effect;
95 transform.theta -= (theta_effect
96 * (rate * std::exp(-eta_decay * time))
97 + last_theta_effect * momentum_alpha)
98 / (100.0 * n); // this makes it work idk
99 last_dx_effect = dx_effect;
100 last_dy_effect = dy_effect;
101 last_theta_effect = theta_effect;
102
103 // compute new cost
104 current_cost = 0;
105 for (size_t i = 0; i < n; i++) {
106 double o_x = (a[i][0] - a_cm_x);
107 double o_y = (a[i][1] - a_cm_y);
108 double a_rot_x = std::cos(transform.theta) * o_x
109 - std::sin(transform.theta) * o_y + a_cm_x;
110 double a_rot_y = std::sin(transform.theta) * o_x
111 + std::cos(transform.theta) * o_y + a_cm_y;
112 double cx = a_rot_x + transform.dx - b[pair[i]][0];
113 double cy = a_rot_y + transform.dy - b[pair[i]][1];
114 current_cost += (cx * cx + cy * cy) / n;
115 }
116
117 time++;
118 }
119 };
