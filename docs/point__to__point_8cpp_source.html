20 void iterate(const std::vector<icp::Point>& a,
21 const std::vector<icp::Point>& b) override {
22 size_t n = a.size();
23 size_t m = b.size();
24 pair.resize(n);
25 dist.resize(n);
26
27 // compute center of mass of a
28 double a_cm_x = 0;
29 double a_cm_y = 0;
30 for (size_t i = 0; i < n; i++) {
31 a_cm_x += a[i].x;
32 a_cm_y += a[i].y;
33 }
34 a_cm_x /= n;
35 a_cm_y /= n;
36
37 t.cx = a_cm_x;
38 t.cy = a_cm_y;
39
40 // point-to-point matching O(n^2)
41 for (size_t i = 0; i < n; i++) {
42 dist[i] = std::numeric_limits<double>::infinity();
43 for (size_t j = 0; j < m; j++) {
44 double alt = std::hypot(a[i].x - b[j].x, a[i].y - b[j].y);
45 if (alt < dist[i]) {
46 pair[i] = j;
47 dist[i] = alt;
48 }
49 }
50 }
51
52 // modified gradient descent
53 double dx_effect = 0;
54 double dy_effect = 0;
55 double theta_effect = 0;
56 for (size_t i = 0; i < n; i++) {
57 double index_dist = ((double)i - pair[i]) / 200;
58 double weight = 1 + 20 * std::exp(-index_dist + index_dist);
59 double o_x = (a[i].x - a_cm_x);
60 double o_y = (a[i].y - a_cm_y);
61 double a_rot_x = std::cos(t.theta) * o_x
62 - std::sin(t.theta) * o_y + a_cm_x;
63 double a_rot_y = std::sin(t.theta) * o_x
64 + std::cos(t.theta) * o_y + a_cm_y;
65 double b_x = b[pair[i]].x;
66 double b_y = b[pair[i]].y;
67 dx_effect += (a_rot_x + t.dx - b_x) * 2 / n * weight;
68 dy_effect += (a_rot_y + t.dy - b_y) * 2 / n * weight;
69 // if (dx_effect > 10000) {
70 // Log << a_rot_x << ' ' << t.dx << ' ' << b_x << '\n';
71 // std::exit(1);
72 // }
73 theta_effect += (b_x * o_x * std::sin(t.theta)
74 + b_x * o_y * std::cos(t.theta)
75 - b_y * o_x * std::cos(t.theta)
76 + b_y * o_y * std::sin(t.theta)
77 - a_cm_x * o_x * std::sin(t.theta)
78 - a_cm_x * o_y * std::cos(t.theta)
79 + a_cm_y * o_x * std::cos(t.theta)
80 - a_cm_y * o_y * std::sin(t.theta)
81 - o_x * t.dx * std::sin(t.theta)
82 + o_x * t.dy * std::cos(t.theta)
83 - o_y * t.dx * std::cos(t.theta)
84 - o_y * t.dy * std::sin(t.theta))
85 * 2 / n * weight;
86 }
87 // Log << "dx_effect * rate = " << (dx_effect * rate) << '\n';
88 // Log << "dy_effect * rate = " << (dy_effect * rate) << '\n';
89 // Log << "dt_effect * rate = " << (theta_effect * rate) << '\n';
90 // std::exit(1);
91 t.dx -= dx_effect * rate + momentum_alpha * last_dx_effect;
92 t.dy -= dy_effect * rate + momentum_alpha * last_dy_effect;
93 t.theta -= (theta_effect * (rate * std::exp(-eta_decay * time))
94 + last_theta_effect * momentum_alpha)
95 / (100.0 * n); // this makes it work idk
96 last_dx_effect = dx_effect;
97 last_dy_effect = dy_effect;
98 last_theta_effect = theta_effect;
99
100 // compute new cost
101 current_cost = 0;
102 for (size_t i = 0; i < n; i++) {
103 double o_x = (a[i].x - a_cm_x);
104 double o_y = (a[i].y - a_cm_y);
105 double a_rot_x = std::cos(t.theta) * o_x
106 - std::sin(t.theta) * o_y + a_cm_x;
107 double a_rot_y = std::sin(t.theta) * o_x
108 + std::cos(t.theta) * o_y + a_cm_y;
109 double cx = a_rot_x + t.dx - b[pair[i]].x;
110 double cy = a_rot_y + t.dy - b[pair[i]].y;
111 current_cost += (cx * cx + cy * cy) / n;
112 }
113
114 time++;
115 }
116 };
