36 void iterate(const std::vector<icp::Point>& a,

38 /* Matching Step: match closest points.
39
40 Sources:
41 https://arxiv.org/pdf/2206.06435.pdf
42 https://web.archive.org/web/20220615080318/https://www.cs.technion.ac.il/~cs236329/tutorials/ICP.pdf

47 for (size_t i = 0; i < n; i++) {
48 dist[i] = std::numeric_limits<double>::infinity();

50 // Point-to-point matching
51 double dist_ij = std::hypot(b[i].x - a[i].x,
52 b[i].y - a[i].y);
53
54 if (dist_ij < dist[i]) {
55 dist[i] = dist_ij;
56 pair[i] = j;
57 }
58 }
59 }

61 /* Transformation Step: */
62 }
