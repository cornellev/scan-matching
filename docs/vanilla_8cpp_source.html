27 void iterate() override {
28 const size_t n = a.size();
29 const size_t m = b.size();
30
31 for (size_t i = 0; i < n; i++) {
32 a_rot[i] = transform.rotation * a[i];
33 }
34
35 /*
36 #step
37 Matching Step: match closest points.
38
39 Sources:
40 https://arxiv.org/pdf/2206.06435.pdf
41 https://web.archive.org/web/20220615080318/https://www.cs.technion.ac.il/~cs236329/tutorials/ICP.pdf
42 https://en.wikipedia.org/wiki/Iterative_closest_point
43 https://courses.cs.duke.edu/spring07/cps296.2/scribe_notes/lecture24.pdf
44 -> use k-d tree
45 */
46 for (size_t i = 0; i < n; i++) {
47 matches[i].sq_dist = std::numeric_limits<double>::infinity();
48 for (size_t j = 0; j < m; j++) {
49 // Point-to-point matching
50 double dist_ij = (b[j] - a_rot[i]).squaredNorm();
51
52 if (dist_ij < matches[i].sq_dist) {
53 matches[i].sq_dist = dist_ij;
54 matches[i].pair = j;
55 }
56 }
57 }
58
59 /*
60 #step
61 Transformation Step: determine optimal transformation.
62
63 The translation vector is determined by the displacement between
64 the centroids of both point clouds. The rotation matrix is
65 calculated via singular value decomposition.
66
67 Sources:
68 https://courses.cs.duke.edu/spring07/cps296.2/scribe_notes/lecture24.pdf
69 */
70 transform.translation = b_cm - transform.rotation * a_cm;
71
72 Matrix N{};
73 for (size_t i = 0; i < n; i++) {
74 N += a[i] * b[matches[i].pair].transpose();
75 }
76 auto svd = N.jacobiSvd(Eigen::ComputeFullU | Eigen::ComputeFullV);
77 const Matrix U = svd.matrixU();
78 const Matrix V = svd.matrixV();
79 transform.rotation = V * U.transpose();
80 }
81 };
