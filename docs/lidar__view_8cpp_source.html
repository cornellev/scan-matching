33void LidarView::construct_instance() {
34 std::random_device rd;
35 std::mt19937 gen(rd());
36 std::uniform_real_distribution<> dis(-sim_config::perturbation_range,
37 sim_config::perturbation_range);
38
39 const double x_unshift = sim_config::n * sim_config::scale / 2;
40 const double y_unshift = sim_config::slope * x_unshift;
41
42 // Create a line of points for the wall
43 for (int i = 0; i < sim_config::n; ++i) {
44 double x = i * sim_config::scale;
45 double y = ((std::sin(i / 10.0) + 1) * sim_config::scale * 100)
46 + sim_config::intercept + dis(gen); // now this is ugly fix
47 wall[i] = {x, y};
48 }
49
50 // Sample the first 3/4th and the last 3/4th as our LiDAR scans
51 source.assign(wall.begin(), wall.begin() + sim_config::n * 3 / 4);
52 destination.assign(wall.begin() + sim_config::n - (sim_config::n * 3 / 4),
53 wall.end());
54 double shift_because_scans_same_start = destination[0].x - source[0].x;
55
56 // Separate the scans to simulate robot movement
57 for (icp::Point& point: source) {
58 point.x += sim_config::window_width / 2 - sim_config::x_displace / 2
59 - x_unshift;
60 point.y += sim_config::window_height / 2 - sim_config::y_displace / 2
61 - y_unshift;
62 }
63
64 double dest_move_x = sim_config::window_width / 2
65 + sim_config::x_displace / 2 - x_unshift;
66 double dest_move_y = sim_config::window_height / 2
67 + sim_config::y_displace / 2 - y_unshift;
68
69 icp::Transform com_rot(0, 0, sim_config::angle_displace);
70 com_rot.cm = icp::get_center_of_mass(source);
71 for (icp::Point& point: source) {
72 double new_x = com_rot.transform_x(point.x, point.y);
73 double new_y = com_rot.transform_y(point.x, point.y);
74 point.x = new_x + sim_config::x_delta + shift_because_scans_same_start;
75 point.y = new_y;
76 }
77
78 for (icp::Point& point: destination) {
79 point.x += dest_move_x + sim_config::x_delta;
80 point.y += dest_move_y;
81 }
82
83 for (icp::Point& point:
84 wall) { // todo: fix this and LidarView to be better code. essentially
85 // you are also shifting the wall to act as gray dots
86 // underneath the destination so you see how close you are to
87 // where you should be. if it works it works but should clean
88 point.x += dest_move_x + sim_config::x_delta;
89 point.y += dest_move_y;
90 }
91}
