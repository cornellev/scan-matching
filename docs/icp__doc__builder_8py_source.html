14 def extract(self, file, contents):
15 pattern = r'/\*\s*(#step|#name|#desc)(.*?)\*/'
16 comments = re.findall(pattern, contents, re.DOTALL)
17 if not comments:
18 return
19 md_filename = 'icp_' + change_extension(file, 'md')
20 with open(self.dir + '/' + md_filename, 'w') as md_file:
21 step_cnt = 1
22 for (kind, comment) in comments:
23 comment_parts = comment.strip().split('Sources:', 1)
24 comment_text = re.sub(r'\n *', '\n', comment_parts[0])
25 sources = re.findall(r'https?://[^\s]+', comment_parts[1]) if len(comment_parts) > 1 else []
26 self.all_sources.update(sources)
27 if kind == '#name':
28 md_file.write(f"\page {comment_text.lower()}_icp {comment_text} ICP\n")
29 elif kind == '#step':
30 lines = comment_text.splitlines()
31 first_line_parts = lines[0].split(':', 1)
32 if len(first_line_parts) == 2:
33 lines[0] = f"**{first_line_parts[0]}**:{first_line_parts[1]}"
34 else:
35 lines[0] = f"**{first_line_parts[0]}**"
36 step_text = '\n'.join(' ' + line for line in lines)
37 md_file.write(f"\n{step_cnt}. {step_text}\n")
38 step_cnt += 1
39 if sources:
40 md_file.write(' Sources: \n')
41 for source in sources:
42 md_file.write(f" - {source}\n")
43 md_file.write('\n')
44 elif kind == '#desc':
45 md_file.write(f"\n{comment_text}\n")
